### 9.3 BilletttypeVelger

```typescript
interface BilletttypeVelgerProps {
  ticketTypes: TicketType[]
  selectedTickets: Array<{ typeId: string; quantity: number }>
  onTicketChange: (typeId: string, quantity: number) => void
  maxPerOrder?: number
}

function BilletttypeVelger({
  ticketTypes,
  selectedTickets,
  onTicketChange,
  maxPerOrder = 10
}: BilletttypeVelgerProps) {
  const totalSelected = selectedTickets.reduce((sum, t) => sum + t.quantity, 0)
  
  return (
    <div className="space-y-4">
      {ticketTypes.map(type => {
        const selected = selectedTickets.find(t => t.typeId === type.id)
        const quantity = selected?.quantity || 0
        const canIncrease = quantity < type.maxPerOrder && 
                           totalSelected < maxPerOrder &&
                           quantity < type.availableQuantity
        
        return (
          <div key={type.id} className="flex items-center justify-between p-4 border rounded-lg">
            <div>
              <p className="font-medium">{type.name}</p>
              {type.description && (
                <p className="text-sm text-stone-500">{type.description}</p>
              )}
            </div>
            <div className="flex items-center gap-4">
              <span className="font-medium">
                {type.price === 0 ? 'Gratis' : `${type.price} kr`}
              </span>
              <div className="flex items-center gap-2">
                <Button 
                  size="sm" 
                  variant="outline"
                  onClick={() => onTicketChange(type.id, Math.max(0, quantity - 1))}
                  disabled={quantity === 0}
                >
                  -
                </Button>
                <span className="w-8 text-center">{quantity}</span>
                <Button 
                  size="sm" 
                  variant="outline"
                  onClick={() => onTicketChange(type.id, quantity + 1)}
                  disabled={!canIncrease}
                >
                  +
                </Button>
              </div>
            </div>
          </div>
        )
      })}
    </div>
  )
}
```
